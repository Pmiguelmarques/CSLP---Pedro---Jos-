62 static void new_distance(char *name1,char *name2,int distance,int special)
63 {
64  int i,j;
65 
66  i = find_city(name1);
67  j = find_city(name2);
68  if(i < 0 || j < 0 || i == j || distance <= 0 || distance >= 1000000)
69  {
70  fprintf(stderr,"Bad info (name1=%s, name2=%s, distance=%d\n",name1,name2,distance);
71  exit(1);
72  }
73  if(special == 0)
74  cities[i].distance[j] = cities[j].distance[i] = distance;
75  else
76  {
77  int c = strcoll(cities[i].name,cities[j].name);
78  if(c < 0)
79  {
80  cities[i].distance[j] = round(0.8 * (double)distance);

83  else
84  {
